from django.forms import ModelForm, HiddenInput
from .models import Job, Comment

class JobForm(ModelForm):
    """
    This class is inherited from ModelForm of Django and designed for wrapping 
    the information of a newly created job item or the updated information 
    of an existed job item. 
    
    Attributes
    ----------
    model : Job
        The target model where this form points to.
    fields : list
        Include basic editable information for a job item.
    """
    # use class name Meta
    class Meta: 
        model = Job
        # fields = "__all__" # include every edittable fields in Job
        fields = [
            "title",
            "description",
            "location",
            "department",
            # "tags",
        ]

def _media(self):
    # Get the media property of the superclass, if it exists
    sup_cls = super(cls, self)
    try:
        base = sup_cls.media
    except AttributeError:
        base = Media()

    # Get the media definition for this class
    definition = getattr(cls, 'Media', None)
    if definition:
        extend = getattr(definition, 'extend', True)
        if extend:
            if extend is True:
                m = base
            else:
                m = Media()
                for medium in extend:
                    m = m + base[medium]
            return m + Media(definition)
        return Media(definition)
    return base
